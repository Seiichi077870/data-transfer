GROUP BY 業者名

フェーズ4：分析・選定
sql

-- 業者別実績サマリー
SELECT 
    v.業者名,
    COUNT(p.工事ID) as 受注件数,
    AVG(p.品質評価) as 平均品質,
    AVG(p.工期遅延日数) as 平均遅延,
    SUM(p.契約金額) as 総受注額
FROM vendors v
LEFT JOIN performance p ON v.業者ID = p.業者ID
GROUP BY v.業者名
ORDER BY 平均品質 DESC, 平均遅延 ASC;

-- 工事種別×業者の実績
SELECT 
    pr.工事種別,
    v.業者名,
    COUNT(*) as 実績件数,
    AVG(pe.品質評価) as 平均評価
FROM projects pr
JOIN performance pe ON pr.工事ID = pe.工事ID
JOIN vendors v ON pe.業者ID = v.業者ID
GROUP BY pr.工事種別, v.業者名
HAVING COUNT(*) >= 3
ORDER BY pr.工事種別, 平均評価 DESC;
